# Operators

```rust
// Arithmetic
let a = 10 + 5;   // 15
let b = 10 - 5;   // 5
let c = 10 * 5;   // 50
let d = 10 / 3;   // 3 (integer division!)
let e = 10 % 3;   // 1 (remainder)

// Comparison
10 == 10   // true
10 != 5    // true
10 > 5     // true
10 < 5     // false
10 >= 10   // true
10 <= 9    // false

// Logical
true && false   // false
true || false   // true
!true           // false

// Bitwise
0b1010 & 0b1100   // AND  → 0b1000
0b1010 | 0b1100   // OR   → 0b1110
0b1010 ^ 0b1100   // XOR  → 0b0110
!0b1010            // NOT
0b1010 << 1        // left shift
0b1010 >> 1        // right shift
```
